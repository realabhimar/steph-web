    <script>
        function checkAnswer() {
            // Get the selected radio button value
            const selectedAnswer = document.querySelector('input[name="answer"]:checked');
            const resultDiv = document.getElementById("result");

            // If no answer is selected
            if (!selectedAnswer) {
                resultDiv.textContent = "Please select an answer!";
                return;
            }

            // Check if the selected answer is correct
            if (selectedAnswer.value === "He is GOATed") {
                resultDiv.textContent = "Correct Answer!";
                resultDiv.style.color = "green";
            } else {
                resultDiv.textContent = "Sorry to hear you're going through that. I mean, thinking Steph Curry is bad? That's a tough spot to be in.I can only imagine the kind of basketball trauma you must have experienced to think that a two-time NBA champion and two-time NBA MVP is 'bad'. My condolences, really.But hey, everyone's entitled to their opinion, no matter how... misinformed. Maybe we can watch some of his games together and, you know, enlighten you?";
                resultDiv.style.color = "red";
            }
        }
    </script>
